\section{Functional requirements}

\makeatletter
\def\requirement#1#2{\begingroup
   \def\@currentlabelname{#2}%
   \label{#1}\endgroup
  #2}
\makeatother

\def \reqref#1 {
  \ref{#1}. \nameref{#1}.
}

The prototype must do the following:

\begin{enumerate}
\item \requirement{req:search}{Search for a video};

\item \requirement{req:play}{Play a video};

\item \requirement{req:pause}{Pause a video};

\item \requirement{req:seek}{Seek in a video};

\end{enumerate}
